import pandas as pd
from pmdarima import auto_arima
import matplotlib.pyplot as plt

# Load data
df = pd.read_csv('/Users/vijaylathwal/Desktop/project/healthcare  sem 4/healthcare-project-covid/dataset/clean_df.csv')

# Convert 'date' to datetime
df['date'] = pd.to_datetime(df['date'])

# Group by date and sum total_cases
total_cases = df.groupby('date')['total_cases'].sum()


# ARIMA model for forecasting
model = auto_arima(total_cases, seasonal=True, m=7)

# Forecast future values
forecast = model.predict(n_periods=30)

from sklearn.decomposition import PCA
from sklearn.preprocessing import StandardScaler
import matplotlib.pyplot as plt

# Load data
df = pd.read_csv('/Users/vijaylathwal/Desktop/project/healthcare  sem 4/healthcare-project-covid/dataset/clean_df.csv')


# Selecting relevant columns for PCA
pca_columns = ['total_cases_per_million', 'people_vaccinated_per_hundred']
pca_data = df[pca_columns]

# Standardizing the data
scaler = StandardScaler()
pca_data_scaled = scaler.fit_transform(pca_data)

# Applying PCA
pca = PCA(n_components=2)
pca_result = pca.fit_transform(pca_data_scaled)

# Explained variance ratio
explained_variance_ratio = pca.explained_variance_ratio_

# Plotting the PCA result
plt.figure(figsize=(10, 6))
plt.scatter(pca_result[:, 0], pca_result[:, 1], alpha=0.7)
plt.xlabel('Principal Component 1')
plt.ylabel('Principal Component 2')
plt.title('PCA of COVID-19 Data')
plt.show(), explained_variance_ratio

from sklearn.cluster import KMeans

# Determining the optimal number of clusters using the Elbow method
wcss = []  # Within-cluster sum of squares
for i in range(1, 11):
    kmeans = KMeans(n_clusters=i, init='k-means++', max_iter=300, n_init=10, random_state=0)
    kmeans.fit(pca_result)
    wcss.append(kmeans.inertia_)

# Plotting the results onto a line graph to observe the 'elbow'
plt.figure(figsize=(10, 6))
plt.plot(range(1, 11), wcss)
plt.title('Elbow Method for Optimal Number of Clusters')
plt.xlabel('Number of clusters')
plt.ylabel('WCSS')
plt.show()
